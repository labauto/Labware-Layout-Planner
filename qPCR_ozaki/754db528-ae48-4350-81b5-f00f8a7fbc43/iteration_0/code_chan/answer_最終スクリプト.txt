    # Pipettes

    pcr_reverse_primer = reagent_rack.wells_by_name()['A3']  # PCR Reverse Primer (10 μM) at position 3
    rox_reference_dye = reagent_rack.wells_by_name()['A4']  # ROX Reference Dye (50×) at position 4
    template = reagent_rack.wells_by_name()['A5']  # Template at position 5
    sterile_water = reagent_rack.wells_by_name()['A6']  # Sterile water at position 6

    primer_R_10uM = reagent_rack.wells_by_name()['B2']  # 10 μM primer_R at position 11

    water = sample_rack.wells_by_name()['A2']  # Water at position 8

    # 96 well PCR plate placed in slot 9
    pcr_plate = protocol.load_labware('nest_96_wellplate_100ul_pcr_full_skirt', '9')  # 96 well PCR plate at position 9

    # Protocol Steps
    # Example: Prepare master mix and distribute to PCR plate

    # Create master mix in a new tube (e.g., in reagent rack well B4)
    master_mix = reagent_rack.wells_by_name()['B4']

    tb_green_premix_volume = 10.0  # μL

    rox_dye_volume = 0.4           # μL
    sterile_water_volume = 6.8     # μL

    # Number of reactions
    num_reactions = 10  # Adjust as necessary

    total_tb_green_premix = tb_green_premix_volume * num_reactions
    total_forward_primer = forward_primer_volume * num_reactions
    total_reverse_primer = reverse_primer_volume * num_reactions
    total_rox_dye = rox_dye_volume * num_reactions
    total_sterile_water = sterile_water_volume * num_reactions

    # Transfer TB Green Premix Ex Taq GC (2×)
    p300.transfer(total_tb_green_premix, tb_green_premix, master_mix.top(), new_tip='always')

    # Transfer PCR Forward Primer (10 μM)
    p10.transfer(total_forward_primer, pcr_forward_primer, master_mix.top(), new_tip='always')

    p10.transfer(total_reverse_primer, pcr_reverse_primer, master_mix.top(), new_tip='always')

    # Distribute master mix to PCR plate

    p300.distribute(master_mix_volume_per_well, master_mix, pcr_plate.wells()[:num_reactions], new_tip='always')

    # Add template DNA to each well
    template_volume = 2.0  # μL per reaction
    for well in pcr_plate.wells()[:num_reactions]:
        p10.transfer(template_volume, template, well, new_tip='always')
